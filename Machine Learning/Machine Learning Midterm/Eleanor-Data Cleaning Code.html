import pandas as pd
from scipy.stats import linregress

# Step 1: Create an empty list to hold results
summary_data = []

# Step 2: Loop through each country and calculate stats
for country, group in df.groupby("Country"):
    group = group.sort_values("Year")
    start_val = group.iloc[0]["Deaths_due_to_Outdoor_Air_Pollution"]
    end_val = group.iloc[-1]["Deaths_due_to_Outdoor_Air_Pollution"]
    avg_val = group["Deaths_due_to_Outdoor_Air_Pollution"].mean()
    
    # Calculate % change and slope of trend line
    pct_change = ((end_val - start_val) / start_val * 100) if start_val != 0 else None
    
    slope, intercept, r_value, p_value, std_err = linregress(
        group["Year"], group["Deaths_due_to_Outdoor_Air_Pollution"]
    )
    
    summary_data.append({
        "Country": country,
        "Start_Deaths_1990": round(start_val, 2),
        "End_Deaths_2019": round(end_val, 2),
        "Average_Deaths": round(avg_val, 2),
        "Absolute_Change": round(end_val - start_val, 2),
        "Percent_Change": round(pct_change, 2) if pct_change is not None else None,
        "Trend_Slope": round(slope, 2),  # annual change in deaths
        "R_squared": round(r_value ** 2, 3)  # goodness of fit
    })

# Step 3: Convert results to a DataFrame
summary_df = pd.DataFrame(summary_data)

# Step 4: Sort by Trend_Slope or Percent_Change (optional)
summary_df = summary_df.sort_values(by="Trend_Slope", ascending=False)

# Step 5: Display first few rows

display(summary_df.head(10))
